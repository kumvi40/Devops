def list_running_hosts():    
	cmd = "vagrant status --machine-readable"    
	status = subprocess.check_output(cmd.split()).rstrip()    
	hosts = []    
	for line in status.split('\n'):        
		(_, host, key, value) = line.split(',')        
		if key == 'state' and value == 'running':            
			hosts.append(host)    
	return hosts
